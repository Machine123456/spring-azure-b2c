select pi.paisorigem, m.simbolo as moedacod, m.descricao as moeda, c.nome as nomeprojecto, p1.nomereprop as nomebeneficiario, t1.paissimbolo as paisbeneficiariocod,t1.descricao as paisbeneficiario, p2.nomereprop as nomerequerente,t2.paissimbolo as paisreqcod, t2.descricao as paisrequerente, c.nomerepres as nomerepresentante, o.valor as montante, ld.datainsercao as prazoutilizacao, pi.montantedesembolso as montantedesembolso, pi.datavalordesembolso as prazodesembolso, c.datainsercao as datacontrato, ld.datainsercao as dataemissaold, pi.datavalordesembolso as dataemissaopi, ld.datainsercao+180 as datavalidadeld, pi.datavalordesembolso+180 as datavalidadepi,pi.paisorigem from dcc_cont_investimento c left join dcc_operacao_cambial o on o.contrato = c.contrato left join dcc_moedas m on c.moeda=m.id_moedas left join dcc_lucro_distribuido ld on o.id_oper_camb = ld.operacaocambial left join dcc_pag_investimento pi on o.id_oper_camb = pi.operacaocambial left join dcc_proponente p1 on p1.id_proponente = ld.proponente left join dcc_proponente p2 on p2.id_proponente = pi.proponente left join dcc_paises t1 on p1.paisprop = t1.id_paises left join dcc_paises t2 on p2.paisprop = t2.id_paises where (pi.paisorigem &lt;> 24 or pi.paisorigem is null) and o.id_oper_camb = #v_operacaoCambialId
